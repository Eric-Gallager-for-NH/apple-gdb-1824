Example: Walking a Tree

    struct tree *
    find (struct tree *tree, int key)
    {
    if (! tree)
    return 0;
    if (key < tree->key)
    return find (tree->left, key);
    else if (key > tree->key)
    return find (tree->right, key);
    else
    return tree;
    }
